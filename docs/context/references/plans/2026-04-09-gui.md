# Squid-Tools GUI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the thin PyQt5 GUI shell over the foundation core library: well plate/region selector, single FOV viewer, napari mosaic view, processing tabs, controls panel, log panel, and embeddable widget.

**Architecture:** Each GUI component is a QWidget that calls core library APIs. No business logic in the GUI layer. The main window composes all panels. Processing tabs are auto-generated from the plugin registry's Pydantic models. Region selector reads `acquisition.yaml` `widget_type` to choose wellplate grid or region dropdown.

**Tech Stack:** PyQt5, napari, superqt, numpy, the squid_tools.core foundation (already built)

**Spec:** `docs/superpowers/specs/2026-04-07-squid-tools-design.md` (GUI Architecture section)

**Foundation dependency:** `squid_tools.core` (data_model, readers, pipeline, registry, sidecar, cache)

```python
from __future__ import annotations

from PyQt5.QtWidgets import QHBoxLayout, QLabel, QWidget


class LogPanel(QWidget):
    """Status bar showing log messages, GPU info, and memory usage."""

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._status_label = QLabel("Ready")
        self._gpu_label = QLabel("GPU: detecting...")
        self._memory_label = QLabel("Mem: --")
        self._log_label = QLabel("")

        layout = QHBoxLayout(self)
        layout.setContentsMargins(4, 2, 4, 2)
        layout.addWidget(self._status_label, stretch=2)
        layout.addWidget(self._log_label, stretch=3)
        layout.addWidget(self._gpu_label, stretch=2)
        layout.addWidget(self._memory_label, stretch=1)

        self._detect_gpu()

    def log(self, message: str) -> None:
        """Display a log message."""
        self._log_label.setText(message)

    def text(self) -> str:
        """Return current log text."""
        return self._log_label.text()

    def set_status(self, status: str) -> None:
        """Set the status label."""
        self._status_label.setText(status)

    def status_text(self) -> str:
        """Return current status text."""
        return self._status_label.text()

    def set_gpu_info(self, gpu_name: str) -> None:
        """Set GPU info label."""
        self._gpu_label.setText(f"GPU: {gpu_name}")

    def gpu_text(self) -> str:
        """Return current GPU text."""
        return self._gpu_label.text()

    def set_memory_info(self, used_gb: float, total_gb: float) -> None:
        """Set memory usage label."""
        self._memory_label.setText(f"Mem: {used_gb:.1f}/{total_gb:.0f} GB")

    def memory_text(self) -> str:
        """Return current memory text."""
        return self._memory_label.text()

    def _detect_gpu(self) -> None:
        """Detect GPU at startup. Non-blocking."""
        try:
            import cupy
            device = cupy.cuda.Device(0)
            name = device.attributes["DeviceName"] if hasattr(device, "attributes") else "CUDA GPU"
            self.set_gpu_info(str(name))
        except Exception:
            self.set_gpu_info("CPU only")
```

```python
"""Left controls panel: view toggle, border overlay, layer controls."""

from __future__ import annotations

from PyQt5.QtCore import pyqtSignal
from PyQt5.QtWidgets import QCheckBox, QLabel, QPushButton, QVBoxLayout, QWidget


class ControlsPanel(QWidget):
    """Left panel with view mode toggle, border overlay, and layer controls."""

    view_mode_changed = pyqtSignal(str)  # "fov" or "mosaic"
    borders_toggled = pyqtSignal(bool)

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._view_mode = "fov"

        layout = QVBoxLayout(self)
        layout.setContentsMargins(4, 4, 4, 4)

        # View toggle
        layout.addWidget(QLabel("View"))
        self.view_toggle_button = QPushButton("FOV View")
        self.view_toggle_button.setToolTip("Toggle between single FOV and mosaic view")
        self.view_toggle_button.setCheckable(True)
        self.view_toggle_button.clicked.connect(self._on_view_toggle)
        layout.addWidget(self.view_toggle_button)

        # Borders
        layout.addWidget(QLabel("Overlay"))
        self.borders_checkbox = QCheckBox("Show FOV Borders")
        self.borders_checkbox.setToolTip("Show/hide FOV border rectangles on mosaic")
        self.borders_checkbox.setChecked(True)
        self.borders_checkbox.toggled.connect(self.borders_toggled.emit)
        layout.addWidget(self.borders_checkbox)

        # Layers
        layout.addWidget(QLabel("Layers"))
        self.original_checkbox = QCheckBox("Original")
        self.original_checkbox.setChecked(True)
        self.original_checkbox.setToolTip("Toggle visibility of original data layer")
        layout.addWidget(self.original_checkbox)

        self.processed_checkbox = QCheckBox("Processed")
        self.processed_checkbox.setChecked(True)
        self.processed_checkbox.setToolTip("Toggle visibility of processed result layer")
        layout.addWidget(self.processed_checkbox)

        layout.addStretch()

    def current_view_mode(self) -> str:
        """Return current view mode: 'fov' or 'mosaic'."""
        return self._view_mode

    def _on_view_toggle(self) -> None:
        if self._view_mode == "fov":
            self._view_mode = "mosaic"
            self.view_toggle_button.setText("Mosaic View")
        else:
            self._view_mode = "fov"
            self.view_toggle_button.setText("FOV View")
        self.view_mode_changed.emit(self._view_mode)
```

```python
"""Right panel: well plate grid or region dropdown.

Auto-detects wellplate vs tissue from Acquisition.mode.
Wellplate mode shows a clickable grid of well buttons.
Tissue/flexible mode shows a dropdown list of region IDs.
"""

from __future__ import annotations

from typing import TYPE_CHECKING

from PyQt5.QtCore import pyqtSignal
from PyQt5.QtWidgets import (
    QComboBox,
    QGridLayout,
    QLabel,
    QPushButton,
    QStackedWidget,
    QVBoxLayout,
    QWidget,
)

if TYPE_CHECKING:
    from squid_tools.core.data_model import Acquisition


class RegionSelector(QWidget):
    """Region selector: wellplate grid or region dropdown."""

    region_selected = pyqtSignal(str)  # region_id

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._wellplate_mode = False
        self._selected_region: str | None = None
        self._well_buttons: dict[str, QPushButton] = {}

        layout = QVBoxLayout(self)
        layout.setContentsMargins(4, 4, 4, 4)
        layout.addWidget(QLabel("Region Selector"))

        self._stack = QStackedWidget()
        layout.addWidget(self._stack)

        # Page 0: wellplate grid
        self._wellplate_widget = QWidget()
        self._wellplate_layout = QGridLayout(self._wellplate_widget)
        self._stack.addWidget(self._wellplate_widget)

        # Page 1: dropdown
        self._dropdown_widget = QWidget()
        dropdown_layout = QVBoxLayout(self._dropdown_widget)
        self._dropdown = QComboBox()
        self._dropdown.setToolTip("Select a region to view")
        self._dropdown.currentTextChanged.connect(self._on_dropdown_changed)
        dropdown_layout.addWidget(self._dropdown)
        dropdown_layout.addStretch()
        self._stack.addWidget(self._dropdown_widget)

    def load_acquisition(self, acq: Acquisition) -> None:
        """Load regions from an acquisition. Auto-detect mode."""
        from squid_tools.core.data_model import AcquisitionMode

        self._wellplate_mode = acq.mode == AcquisitionMode.WELLPLATE
        self._well_buttons.clear()

        if self._wellplate_mode:
            # Clear existing grid
            while self._wellplate_layout.count():
                item = self._wellplate_layout.takeAt(0)
                if item.widget():
                    item.widget().deleteLater()

            # Build grid from region IDs (e.g., "A1", "B2")
            for region_id in sorted(acq.regions.keys()):
                row, col = self._parse_well_position(region_id)
                btn = QPushButton(region_id)
                btn.setToolTip(f"View region {region_id}")
                btn.setFixedSize(40, 30)
                btn.clicked.connect(lambda checked, rid=region_id: self.select_region(rid))
                self._wellplate_layout.addWidget(btn, row, col)
                self._well_buttons[region_id] = btn

            self._stack.setCurrentIndex(0)
        else:
            self._dropdown.clear()
            for region_id in sorted(acq.regions.keys()):
                self._dropdown.addItem(region_id)
            self._stack.setCurrentIndex(1)

    def is_wellplate_mode(self) -> bool:
        """Return True if showing wellplate grid."""
        return self._wellplate_mode

    def select_region(self, region_id: str) -> None:
        """Programmatically select a region."""
        self._selected_region = region_id
        self.region_selected.emit(region_id)

    def selected_region_id(self) -> str | None:
        """Return currently selected region ID."""
        return self._selected_region

    def _on_dropdown_changed(self, text: str) -> None:
        if text:
            self.select_region(text)

    @staticmethod
    def _parse_well_position(region_id: str) -> tuple[int, int]:
        """Parse 'A1' -> (0, 0), 'B2' -> (1, 1), etc. Fallback to (0, idx)."""
        if len(region_id) >= 2 and region_id[0].isalpha() and region_id[1:].isdigit():
            row = ord(region_id[0].upper()) - ord("A")
            col = int(region_id[1:]) - 1
            return row, col
        return 0, 0
```

```python
"""Top processing tabs: one tab per installed plugin.

Parameter widgets are auto-generated from each plugin's Pydantic model.
Each tab has a Run button that emits run_requested(plugin_name).
"""

from __future__ import annotations

from typing import Any

from pydantic import BaseModel
from PyQt5.QtCore import pyqtSignal
from PyQt5.QtWidgets import (
    QDoubleSpinBox,
    QFormLayout,
    QHBoxLayout,
    QPushButton,
    QSpinBox,
    QTabWidget,
    QVBoxLayout,
    QWidget,
)

from squid_tools.core.registry import PluginRegistry
from squid_tools.plugins.base import ProcessingPlugin


class ProcessingTabs(QTabWidget):
    """Tab widget with one tab per registered plugin."""

    run_requested = pyqtSignal(str, object)  # (plugin_name, params_dict)

    def __init__(self, registry: PluginRegistry, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._registry = registry
        self._plugin_tabs: dict[str, _PluginTab] = {}

        for plugin in registry.list_all():
            tab = _PluginTab(plugin, self)
            tab.run_clicked.connect(lambda name=plugin.name: self._on_run(name))
            self.addTab(tab, plugin.name)
            self._plugin_tabs[plugin.name] = tab

    def trigger_run(self, plugin_name: str) -> None:
        """Programmatically trigger a run for a plugin."""
        if plugin_name in self._plugin_tabs:
            self._plugin_tabs[plugin_name].click_run()

    def tab_text(self, index: int) -> str:
        """Return tab text for given index."""
        return self.tabText(index)

    def _on_run(self, plugin_name: str) -> None:
        if plugin_name in self._plugin_tabs:
            params = self._plugin_tabs[plugin_name].get_params()
            self.run_requested.emit(plugin_name, params)


class _PluginTab(QWidget):
    """Single plugin tab with auto-generated parameter widgets and Run button."""

    run_clicked = pyqtSignal()

    def __init__(self, plugin: ProcessingPlugin, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._plugin = plugin
        self._param_widgets: dict[str, QWidget] = {}

        layout = QVBoxLayout(self)

        # Auto-generate parameter widgets from Pydantic model
        params_model = plugin.parameters()
        form = QFormLayout()

        for field_name, field_info in params_model.model_fields.items():
            annotation = field_info.annotation
            default = field_info.default
            widget: QWidget

            if annotation is float or annotation is (float | None):
                spin = QDoubleSpinBox()
                spin.setRange(-1e6, 1e6)
                spin.setDecimals(3)
                if isinstance(default, (int, float)):
                    spin.setValue(float(default))
                spin.setToolTip(field_info.description or f"Parameter: {field_name}")
                widget = spin
            elif annotation is int or annotation is (int | None):
                spin_int = QSpinBox()
                spin_int.setRange(0, 100000)
                if isinstance(default, int):
                    spin_int.setValue(default)
                spin_int.setToolTip(field_info.description or f"Parameter: {field_name}")
                widget = spin_int
            else:
                # Fallback: skip unsupported types
                continue

            self._param_widgets[field_name] = widget
            form.addRow(field_name, widget)

        layout.addLayout(form)

        # Run button
        btn_layout = QHBoxLayout()
        self._run_button = QPushButton("Run")
        self._run_button.setToolTip(f"Run {plugin.name}")
        self._run_button.clicked.connect(self.run_clicked.emit)
        btn_layout.addStretch()
        btn_layout.addWidget(self._run_button)
        layout.addLayout(btn_layout)
        layout.addStretch()

    def click_run(self) -> None:
        """Programmatically click the run button."""
        self._run_button.click()

    def get_params(self) -> dict[str, Any]:
        """Read current parameter values from widgets."""
        params: dict[str, Any] = {}
        for name, widget in self._param_widgets.items():
            if isinstance(widget, QDoubleSpinBox):
                params[name] = widget.value()
            elif isinstance(widget, QSpinBox):
                params[name] = widget.value()
        return params
```

```python
"""Center panel: mosaic view using napari with coordinate-placed tiles.

Each FOV is placed at its (x_mm, y_mm) position via napari translate
transforms. napari handles viewport-based rendering. FOV borders are
shown as a Shapes layer.
"""

from __future__ import annotations

from typing import TYPE_CHECKING

import numpy as np
from PyQt5.QtWidgets import QVBoxLayout, QWidget

if TYPE_CHECKING:
    from squid_tools.core.data_model import Acquisition, Region


class MosaicView(QWidget):
    """Displays tiled FOVs in a mosaic using napari."""

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)

        import napari

        self._viewer = napari.Viewer(show=False)
        self._napari_widget = self._viewer.window._qt_window
        layout.addWidget(self._napari_widget)

        self._border_shapes: list[np.ndarray] = []
        self._show_borders = True

    def load_region(
        self,
        region: Region,
        frames: dict[int, np.ndarray],
        pixel_size_um: float,
    ) -> None:
        """Load FOV tiles for a region into the mosaic.

        Args:
            region: Region with FOV positions.
            frames: Dict mapping fov_index -> 2D numpy array.
            pixel_size_um: Pixel size in microns for coordinate conversion.
        """
        self._viewer.layers.clear()
        self._border_shapes.clear()

        for fov in region.fovs:
            if fov.fov_index not in frames:
                continue

            frame = frames[fov.fov_index]
            # Convert mm to pixels for translate
            y_px = fov.y_mm * 1000 / pixel_size_um
            x_px = fov.x_mm * 1000 / pixel_size_um

            self._viewer.add_image(
                frame,
                name=f"fov_{fov.fov_index}",
                translate=(y_px, x_px),
                blending="translucent",
            )

            # Build border rectangle
            h, w = frame.shape[:2]
            rect = np.array([
                [y_px, x_px],
                [y_px, x_px + w],
                [y_px + h, x_px + w],
                [y_px + h, x_px],
            ])
            self._border_shapes.append(rect)

        if self._show_borders and self._border_shapes:
            self._viewer.add_shapes(
                self._border_shapes,
                shape_type="polygon",
                edge_color="white",
                edge_width=2,
                face_color="transparent",
                name="fov_borders",
            )

        self._viewer.reset_view()

    def set_borders_visible(self, visible: bool) -> None:
        """Show or hide FOV border overlay."""
        self._show_borders = visible
        if "fov_borders" in self._viewer.layers:
            self._viewer.layers["fov_borders"].visible = visible

    def display_processed(
        self, fov_index: int, frame: np.ndarray, y_px: float, x_px: float
    ) -> None:
        """Add a processed result layer for a specific FOV."""
        name = f"processed_fov_{fov_index}"
        if name in self._viewer.layers:
            self._viewer.layers[name].data = frame
        else:
            self._viewer.add_image(
                frame,
                name=name,
                translate=(y_px, x_px),
                colormap="green",
                blending="additive",
            )

    def clear(self) -> None:
        """Remove all layers."""
        self._viewer.layers.clear()
        self._border_shapes.clear()

    @property
    def napari_viewer(self) -> object:
        """Access the underlying napari viewer."""
        return self._viewer
```

```python
Composes all GUI panels: controls (left), viewer/mosaic (center),

class MainWindow(QMainWindow):
    """Squid-Tools main window."""

    def __init__(self, registry: PluginRegistry | None = None) -> None:
        super().__init__()
        self.setWindowTitle("Squid-Tools")
        self.setMinimumSize(1024, 768)

        if registry is None:
            registry = PluginRegistry()

        # Central widget
        central = QWidget()
        self.setCentralWidget(central)
        main_layout = QVBoxLayout(central)
        main_layout.setContentsMargins(0, 0, 0, 0)
        main_layout.setSpacing(0)

        # Top: processing tabs
        self.processing_tabs = ProcessingTabs(registry)
        main_layout.addWidget(self.processing_tabs)

        # Middle: controls | viewer placeholder | region selector
        middle_splitter = QSplitter(Qt.Horizontal)

        self.controls_panel = ControlsPanel()
        self.controls_panel.setMaximumWidth(150)
        middle_splitter.addWidget(self.controls_panel)

        # Viewer placeholder (napari widgets created lazily on data load)
        self._viewer_placeholder = QWidget()
        self._viewer_layout = QVBoxLayout(self._viewer_placeholder)
        self._viewer_layout.setContentsMargins(0, 0, 0, 0)

        self.region_selector = RegionSelector()
        self.region_selector.setMaximumWidth(200)
        middle_splitter.addWidget(self.region_selector)

        middle_splitter.setStretchFactor(0, 0)  # controls: fixed
        middle_splitter.setStretchFactor(1, 1)  # viewer: stretches
        middle_splitter.setStretchFactor(2, 0)  # region: fixed

        main_layout.addWidget(middle_splitter, stretch=1)

        # Bottom: log panel
        self.log_panel = LogPanel()
        self.log_panel.setMaximumHeight(30)
        main_layout.addWidget(self.log_panel)

    window.show()

    main()
```
