> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the placeholder viewer/mosaic widgets with a custom vispy+PySide6 viewer that renders 2D image tiles with GPU acceleration, supports single FOV and mosaic modes, channel/z/t sliders, FOV border overlay, and multi-channel compositing.

**Architecture:** Three files with clear separation: `canvas.py` (vispy SceneCanvas, OpenGL rendering, knows nothing about data or Qt widgets), `data_manager.py` (viewport-aware loading from core/readers via cache, knows nothing about OpenGL), `widget.py` (QWidget composing canvas + sliders, two view modes). The viewer replaces the placeholder QLabel implementations while keeping the same public API that app.py already calls.

**Tech Stack:** vispy (SceneCanvas, Image visual, PanZoomCamera, STTransform, Line visual), PySide6, numpy, existing core/ library

**Spec:** `docs/superpowers/specs/2026-04-09-squid-tools-v1-design.md` (Custom Vispy+PySide6 Viewer section)

`squid_tools/viewer/canvas.py`:
```python
"""Vispy SceneCanvas wrapper for GPU-accelerated tile rendering.

Responsibilities:
- Upload numpy arrays as OpenGL textures
- Pan/zoom via PanZoomCamera
- Multi-channel compositing with colormaps
- FOV border overlay (line geometry)
- Tile placement at pixel coordinates (mosaic mode)

Knows nothing about acquisitions, files, dask, or Qt widgets.
"""

from __future__ import annotations

import numpy as np
import vispy.app
from vispy.gloo import set_state
from vispy.scene import SceneCanvas
from vispy.scene.cameras import PanZoomCamera
from vispy.scene.visuals import Image, Line
from vispy.visuals.transforms import STTransform

# Use PySide6 backend
vispy.app.use_app("pyside6")


def _normalize(data: np.ndarray) -> np.ndarray:
    """Normalize image data to float32 in [0, 1] for vispy."""
    if data.dtype == np.float32 and data.max() <= 1.0:
        return data
    data = data.astype(np.float32)
    dmax = data.max()
    if dmax > 0:
        data = data / dmax
    return data


class VispyCanvas:
    """Wraps a vispy SceneCanvas for 2D microscopy image display.

    Supports single-image mode and tiled mosaic mode.
    """

    def __init__(self) -> None:
        self._canvas = SceneCanvas(keys="interactive", show=False)
        self._view = self._canvas.central_widget.add_view()
        self._view.camera = PanZoomCamera(aspect=1)

        # Enable additive blending for multi-channel compositing
        set_state(blend=True, blend_func=("src_alpha", "one"))

        # Single image (set_image mode)
        self._main_image: Image | None = None

        # Tiles (add_tile mode)
        self._tiles: dict[str, Image] = {}

        # Borders
        self._borders: dict[str, Line] = {}
        self._borders_visible = True

    def set_image(self, data: np.ndarray, cmap: str = "grays", clim: tuple[float, float] | None = None) -> None:
        """Display a single image (single FOV mode). Replaces any existing image."""
        normalized = _normalize(data)
        if clim is None:
            clim = (0.0, 1.0)

        if self._main_image is not None:
            self._main_image.set_data(normalized)
            self._main_image.cmap = cmap
            self._main_image.clim = clim
        else:
            self._main_image = Image(normalized, cmap=cmap, clim=clim, parent=self._view.scene)

    def has_image(self) -> bool:
        """Return True if an image or tiles are displayed."""
        return self._main_image is not None or len(self._tiles) > 0

    def add_tile(
        self,
        data: np.ndarray,
        x: float,
        y: float,
        tile_id: str,
        cmap: str = "grays",
        clim: tuple[float, float] | None = None,
    ) -> None:
        """Add a tile at pixel coordinates (mosaic mode)."""
        normalized = _normalize(data)
        if clim is None:
            clim = (0.0, 1.0)

        if tile_id in self._tiles:
            self._tiles[tile_id].set_data(normalized)
            self._tiles[tile_id].transform = STTransform(translate=(x, y, 0))
        else:
            img = Image(normalized, cmap=cmap, clim=clim, parent=self._view.scene)
            img.transform = STTransform(translate=(x, y, 0))
            self._tiles[tile_id] = img

    def tile_count(self) -> int:
        """Return number of tiles in the mosaic."""
        return len(self._tiles)

    def add_border(
        self,
        x: float,
        y: float,
        width: float,
        height: float,
        border_id: str,
        color: str = "white",
    ) -> None:
        """Add a rectangular FOV border overlay."""
        corners = np.array([
            [x, y],
            [x + width, y],
            [x + width, y + height],
            [x, y + height],
            [x, y],
        ], dtype=np.float32)

        if border_id in self._borders:
            self._borders[border_id].set_data(pos=corners, color=color)
        else:
            line = Line(pos=corners, color=color, width=2, connect="strip", parent=self._view.scene)
            self._borders[border_id] = line

    def border_count(self) -> int:
        """Return number of border overlays."""
        return len(self._borders)

    def set_borders_visible(self, visible: bool) -> None:
        """Show or hide all FOV border overlays."""
        self._borders_visible = visible
        for border in self._borders.values():
            border.visible = visible

    def update_border_color(self, border_id: str, color: str) -> None:
        """Update a border's color (e.g., white -> green after registration)."""
        if border_id in self._borders:
            self._borders[border_id].set_data(color=color)

    def clear_images(self) -> None:
        """Remove all images, tiles, and borders."""
        if self._main_image is not None:
            self._main_image.parent = None
            self._main_image = None

        for tile in self._tiles.values():
            tile.parent = None
        self._tiles.clear()

        for border in self._borders.values():
            border.parent = None
        self._borders.clear()

    def fit_view(self) -> None:
        """Reset camera to fit all content."""
        self._view.camera.set_range()

    def native_widget(self) -> object:
        """Return the native QWidget for embedding in PySide6 layouts."""
        return self._canvas.native

    def get_viewport_bounds(self) -> tuple[float, float, float, float]:
        """Return current viewport as (x_min, y_min, x_max, y_max) in scene coords."""
        rect = self._view.camera.rect
        return (rect.left, rect.bottom, rect.right, rect.top)
```

`squid_tools/viewer/data_manager.py`:
```python
"""Viewport-aware data loading for the viewer.

Loads frames via core/readers, caches via core/cache.
Knows nothing about OpenGL or Qt.
"""

from __future__ import annotations

from pathlib import Path

import numpy as np

from squid_tools.core.cache import MemoryBoundedLRUCache
from squid_tools.core.data_model import Acquisition, FrameKey
from squid_tools.core.readers import detect_reader
from squid_tools.core.readers.base import FormatReader


class ViewportDataManager:
    """Manages data loading for the viewer with caching."""

    def __init__(self, cache_bytes: int = 256 * 1024 * 1024) -> None:
        self.acquisition: Acquisition | None = None
        self._reader: FormatReader | None = None
        self._cache = MemoryBoundedLRUCache(max_bytes=cache_bytes)

    def load(self, path: Path) -> Acquisition:
        """Load an acquisition directory."""
        self._reader = detect_reader(path)
        self.acquisition = self._reader.read_metadata(path)
        self._cache.clear()
        return self.acquisition

    @property
    def pixel_size_um(self) -> float:
        """Pixel size in microns."""
        if self.acquisition is None:
            raise RuntimeError("No acquisition loaded")
        return self.acquisition.objective.pixel_size_um

    def region_ids(self) -> list[str]:
        """List available region IDs."""
        if self.acquisition is None:
            return []
        return list(self.acquisition.regions.keys())

    def get_frame(
        self,
        region: str,
        fov: int,
        z: int = 0,
        channel: int = 0,
        timepoint: int = 0,
    ) -> np.ndarray:
        """Load a single frame with caching."""
        if self._reader is None:
            raise RuntimeError("No acquisition loaded")

        cache_key = f"{region}_{fov}_{z}_{channel}_{timepoint}"
        cached = self._cache.get(cache_key)
        if cached is not None:
            return cached

        key = FrameKey(region=region, fov=fov, z=z, channel=channel, timepoint=timepoint)
        frame = self._reader.read_frame(key)
        self._cache.put(cache_key, frame)
        return frame

    def get_region_frames(
        self,
        region: str,
        z: int = 0,
        channel: int = 0,
        timepoint: int = 0,
    ) -> dict[int, np.ndarray]:
        """Load all FOV frames for a region."""
        if self.acquisition is None:
            raise RuntimeError("No acquisition loaded")
        if region not in self.acquisition.regions:
            raise ValueError(f"Region '{region}' not found")

        frames: dict[int, np.ndarray] = {}
        for fov in self.acquisition.regions[region].fovs:
            frames[fov.fov_index] = self.get_frame(
                region=region, fov=fov.fov_index, z=z, channel=channel, timepoint=timepoint
            )
        return frames
```

```python
"""Viewer widget composing vispy canvas with sliders and view modes.

Two modes:
- FOV: single image display with channel/z/t sliders
- Mosaic: all FOVs placed by coordinates with border overlay

Emits signals for mode changes and FOV clicks.
"""

from __future__ import annotations

from pathlib import Path
from typing import TYPE_CHECKING

import numpy as np
from PySide6.QtCore import Signal
from PySide6.QtWidgets import (
    QHBoxLayout,
    QLabel,
    QSlider,
    QVBoxLayout,
    QWidget,
)
from PySide6.QtCore import Qt

from squid_tools.viewer.canvas import VispyCanvas
from squid_tools.viewer.data_manager import ViewportDataManager

if TYPE_CHECKING:
    from squid_tools.core.data_model import Acquisition


class ViewerWidget(QWidget):
    """Main viewer widget with vispy canvas, sliders, and FOV/mosaic modes."""

    mode_changed = Signal(str)  # "fov" or "mosaic"
    fov_clicked = Signal(str, int)  # region_id, fov_index

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self._mode = "fov"
        self._current_region: str | None = None

        self._canvas = VispyCanvas()
        self._data = ViewportDataManager()

        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)
        layout.setSpacing(0)

        # Canvas
        native = self._canvas.native_widget()
        if native is not None:
            layout.addWidget(native, stretch=1)  # type: ignore[arg-type]

        # Sliders row
        slider_row = QHBoxLayout()
        slider_row.setContentsMargins(4, 2, 4, 2)

        slider_row.addWidget(QLabel("Ch"))
        self.channel_slider = QSlider(Qt.Horizontal)
        self.channel_slider.setRange(0, 0)
        self.channel_slider.setToolTip("Channel")
        self.channel_slider.valueChanged.connect(self._on_slider_changed)
        slider_row.addWidget(self.channel_slider)

        slider_row.addWidget(QLabel("Z"))
        self.z_slider = QSlider(Qt.Horizontal)
        self.z_slider.setRange(0, 0)
        self.z_slider.setToolTip("Z-level")
        self.z_slider.valueChanged.connect(self._on_slider_changed)
        slider_row.addWidget(self.z_slider)

        slider_row.addWidget(QLabel("T"))
        self.t_slider = QSlider(Qt.Horizontal)
        self.t_slider.setRange(0, 0)
        self.t_slider.setToolTip("Timepoint")
        self.t_slider.valueChanged.connect(self._on_slider_changed)
        slider_row.addWidget(self.t_slider)

        layout.addLayout(slider_row)

    def load_acquisition(self, path: Path) -> None:
        """Load an acquisition and configure sliders."""
        acq = self._data.load(path)
        self._configure_sliders(acq)

    def _configure_sliders(self, acq: Acquisition) -> None:
        """Set slider ranges from acquisition metadata."""
        nc = len(acq.channels)
        self.channel_slider.setRange(0, max(0, nc - 1))

        nz = acq.z_stack.nz if acq.z_stack else 1
        self.z_slider.setRange(0, max(0, nz - 1))

        nt = acq.time_series.nt if acq.time_series else 1
        self.t_slider.setRange(0, max(0, nt - 1))

    def display_fov(self, frame: np.ndarray, name: str = "image", cmap: str = "grays") -> None:
        """Display a single FOV frame."""
        self._mode = "fov"
        self._canvas.clear_images()
        self._canvas.set_image(frame, cmap=cmap)
        self._canvas.fit_view()

    def display_mosaic(self, region_id: str, cmap: str = "grays") -> None:
        """Display all FOVs for a region as a mosaic."""
        if self._data.acquisition is None:
            return

        self._mode = "mosaic"
        self._current_region = region_id
        self._canvas.clear_images()

        region = self._data.acquisition.regions.get(region_id)
        if region is None:
            return

        z = self.z_slider.value()
        ch = self.channel_slider.value()
        t = self.t_slider.value()

        frames = self._data.get_region_frames(region=region_id, z=z, channel=ch, timepoint=t)
        pixel_size = self._data.pixel_size_um

        for fov in region.fovs:
            if fov.fov_index not in frames:
                continue
            frame = frames[fov.fov_index]
            # Convert mm to pixels
            x_px = fov.x_mm * 1000 / pixel_size
            y_px = fov.y_mm * 1000 / pixel_size
            self._canvas.add_tile(
                frame, x=x_px, y=y_px, tile_id=f"fov_{fov.fov_index}", cmap=cmap
            )
            # Add border
            h, w = frame.shape[:2]
            self._canvas.add_border(
                x=x_px, y=y_px, width=float(w), height=float(h),
                border_id=f"border_{fov.fov_index}",
            )

        self._canvas.fit_view()

    def display_processed(self, frame: np.ndarray, name: str = "processed", cmap: str = "green") -> None:
        """Overlay a processed result."""
        self._canvas.set_image(frame, cmap=cmap)

    def set_mode(self, mode: str) -> None:
        """Switch between 'fov' and 'mosaic' mode."""
        self._mode = mode
        self.mode_changed.emit(mode)

    def current_mode(self) -> str:
        """Return current view mode."""
        return self._mode

    def set_borders_visible(self, visible: bool) -> None:
        """Toggle FOV border overlay."""
        self._canvas.set_borders_visible(visible)

    def clear(self) -> None:
        """Clear all displayed content."""
        self._canvas.clear_images()

    def _on_slider_changed(self) -> None:
        """Reload current view when a slider changes."""
        if self._mode == "mosaic" and self._current_region and self._data.acquisition:
            self.display_mosaic(self._current_region)
        # FOV mode slider changes handled by controller
```

"""Mosaic view adapter.

Wraps ViewerWidget for the MosaicView API that app.py expects.
"""

from __future__ import annotations

from pathlib import Path
from typing import TYPE_CHECKING

import numpy as np
from PySide6.QtWidgets import QVBoxLayout, QWidget

from squid_tools.viewer.widget import ViewerWidget

if TYPE_CHECKING:
    from squid_tools.core.data_model import Region


class MosaicView(QWidget):
    """Mosaic display using the vispy viewer."""

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)
        self._viewer = ViewerWidget()
        layout.addWidget(self._viewer)

    def load_region(
        self,
        region: Region,
        frames: dict[int, np.ndarray],
        pixel_size_um: float,
    ) -> None:
        # Load data into data manager isn't needed here since frames are passed directly
        self._viewer._canvas.clear_images()
        for fov in region.fovs:
            if fov.fov_index not in frames:
                continue
            frame = frames[fov.fov_index]
            x_px = fov.x_mm * 1000 / pixel_size_um
            y_px = fov.y_mm * 1000 / pixel_size_um
            self._viewer._canvas.add_tile(
                frame, x=x_px, y=y_px, tile_id=f"fov_{fov.fov_index}"
            )
            h, w = frame.shape[:2]
            self._viewer._canvas.add_border(
                x=x_px, y=y_px, width=float(w), height=float(h),
                border_id=f"border_{fov.fov_index}",
            )
        self._viewer._canvas.fit_view()

    def set_borders_visible(self, visible: bool) -> None:
        self._viewer.set_borders_visible(visible)

    def display_processed(
        self, fov_index: int, frame: np.ndarray, y_px: float, x_px: float
    ) -> None:
        self._viewer._canvas.add_tile(
            frame, x=x_px, y=y_px, tile_id=f"processed_{fov_index}", cmap="green"
        )

    def clear(self) -> None:
        self._viewer.clear()

    @property
    def viewer_widget(self) -> ViewerWidget:
        return self._viewer
