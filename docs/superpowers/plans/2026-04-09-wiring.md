```python
"""Application controller: mediates between GUI widgets and core library.

Owns the Acquisition, FormatReader, Pipeline, PluginRegistry, and
SidecarManifest. GUI widgets emit signals, the controller handles them.
All business logic lives here, not in widgets.
"""

from __future__ import annotations

from pathlib import Path
from typing import Any

import numpy as np

from squid_tools.core.data_model import Acquisition, FrameKey
from squid_tools.core.pipeline import Pipeline
from squid_tools.core.readers import detect_reader
from squid_tools.core.readers.base import FormatReader
from squid_tools.core.registry import PluginRegistry
from squid_tools.core.sidecar import SidecarManifest


class AppController:
    """Central controller for squid-tools.

    Manages data loading, frame retrieval, plugin execution, and sidecar output.
    GUI widgets call this controller; it calls the core library.
    """

    def __init__(self, registry: PluginRegistry | None = None) -> None:
        self.registry = registry or PluginRegistry()
        self.acquisition: Acquisition | None = None
        self.pipeline = Pipeline()
        self.sidecar: SidecarManifest | None = None
        self._reader: FormatReader | None = None

    def load_acquisition(self, path: Path) -> Acquisition:
        """Load an acquisition directory. Auto-detects format."""
        self._reader = detect_reader(path)
        self.acquisition = self._reader.read_metadata(path)
        self.sidecar = SidecarManifest(acquisition_path=path)
        return self.acquisition

    def get_frame(
        self,
        region: str,
        fov: int,
        z: int = 0,
        channel: int = 0,
        timepoint: int = 0,
    ) -> np.ndarray:
        """Load a single frame."""
        if self._reader is None:
            raise RuntimeError("No acquisition loaded")
        key = FrameKey(region=region, fov=fov, z=z, channel=channel, timepoint=timepoint)
        return self._reader.read_frame(key)

    def get_region_frames(
        self,
        region: str,
        z: int = 0,
        channel: int = 0,
        timepoint: int = 0,
    ) -> dict[int, np.ndarray]:
        """Load all FOV frames for a region. Returns {fov_index: frame}."""
        if self._reader is None or self.acquisition is None:
            raise RuntimeError("No acquisition loaded")
        if region not in self.acquisition.regions:
            raise ValueError(f"Region '{region}' not found")

        frames: dict[int, np.ndarray] = {}
        for fov in self.acquisition.regions[region].fovs:
            key = FrameKey(
                region=region, fov=fov.fov_index, z=z, channel=channel, timepoint=timepoint
            )
            frames[fov.fov_index] = self._reader.read_frame(key)
        return frames

    def run_plugin(
        self,
        plugin_name: str,
        frame: np.ndarray,
        params: dict[str, Any] | None = None,
    ) -> np.ndarray:
        """Run a plugin on a frame. Returns processed frame."""
        plugin = self.registry.get(plugin_name)
        if plugin is None:
            raise ValueError(f"Plugin '{plugin_name}' not found")

        params_model_cls = plugin.parameters()
        if params:
            params_model = params_model_cls(**params)
        else:
            optical = self.acquisition.optical if self.acquisition else None
            params_model = plugin.default_params(optical)

        return plugin.process(frame, params_model)
```

```python
"""Main application window for Squid-Tools.

Composes all GUI panels: controls (left), viewer/mosaic (center),
region selector (right), processing tabs (top), log panel (bottom).
Wired to AppController for data loading and processing.
"""

from __future__ import annotations

import sys
from pathlib import Path

from PyQt5.QtCore import Qt
from PyQt5.QtWidgets import (
    QAction,
    QApplication,
    QFileDialog,
    QMainWindow,
    QSplitter,
    QStackedWidget,
    QVBoxLayout,
    QWidget,
)

from squid_tools.core.registry import PluginRegistry
from squid_tools.gui.controller import AppController
from squid_tools.gui.controls import ControlsPanel
from squid_tools.gui.log_panel import LogPanel
from squid_tools.gui.processing_tabs import ProcessingTabs
from squid_tools.gui.region_selector import RegionSelector


class MainWindow(QMainWindow):
    """Squid-Tools main window."""

    def __init__(self, registry: PluginRegistry | None = None) -> None:
        super().__init__()
        self.setWindowTitle("Squid-Tools")
        self.setMinimumSize(1024, 768)

        self.controller = AppController(registry)

        # Menu bar
        self._setup_menus()

        # Central widget
        central = QWidget()
        self.setCentralWidget(central)
        main_layout = QVBoxLayout(central)
        main_layout.setContentsMargins(0, 0, 0, 0)
        main_layout.setSpacing(0)

        # Top: processing tabs
        self.processing_tabs = ProcessingTabs(self.controller.registry)
        self.processing_tabs.run_requested.connect(self._on_run_requested)
        main_layout.addWidget(self.processing_tabs)

        # Middle: controls | viewer stack | region selector
        middle_splitter = QSplitter(Qt.Horizontal)

        self.controls_panel = ControlsPanel()
        self.controls_panel.setMaximumWidth(150)
        self.controls_panel.view_mode_changed.connect(self._on_view_mode_changed)
        self.controls_panel.borders_toggled.connect(self._on_borders_toggled)
        middle_splitter.addWidget(self.controls_panel)

        # Viewer stack: page 0 = placeholder, page 1 = single FOV, page 2 = mosaic
        self._viewer_stack = QStackedWidget()
        self._viewer_stack.addWidget(QWidget())  # page 0: placeholder
        middle_splitter.addWidget(self._viewer_stack)

        self.region_selector = RegionSelector()
        self.region_selector.setMaximumWidth(200)
        self.region_selector.region_selected.connect(self._on_region_selected)
        middle_splitter.addWidget(self.region_selector)

        middle_splitter.setStretchFactor(0, 0)
        middle_splitter.setStretchFactor(1, 1)
        middle_splitter.setStretchFactor(2, 0)

        main_layout.addWidget(middle_splitter, stretch=1)

        # Bottom: log panel
        self.log_panel = LogPanel()
        self.log_panel.setMaximumHeight(30)
        main_layout.addWidget(self.log_panel)

        # Lazily created viewers
        self._fov_viewer = None
        self._mosaic_viewer = None

    def _setup_menus(self) -> None:
        menu_bar = self.menuBar()
        file_menu = menu_bar.addMenu("&File")

        open_action = QAction("&Open Acquisition...", self)
        open_action.setShortcut("Ctrl+O")
        open_action.setToolTip("Open a Squid acquisition directory")
        open_action.triggered.connect(self._on_open)
        file_menu.addAction(open_action)

        file_menu.addSeparator()

        quit_action = QAction("&Quit", self)
        quit_action.setShortcut("Ctrl+Q")
        quit_action.triggered.connect(self.close)
        file_menu.addAction(quit_action)

    def _on_open(self) -> None:
        path = QFileDialog.getExistingDirectory(self, "Open Acquisition Directory")
        if path:
            self.open_acquisition(Path(path))

    def open_acquisition(self, path: Path) -> None:
        """Load an acquisition and populate all panels."""
        try:
            acq = self.controller.load_acquisition(path)
            self.region_selector.load_acquisition(acq)
            self.log_panel.set_status(f"Loaded: {path.name}")
            self.log_panel.log(
                f"{acq.format.value} | {len(acq.regions)} region(s) | "
                f"{sum(len(r.fovs) for r in acq.regions.values())} FOVs"
            )

            # Auto-select first region
            if acq.regions:
                first_region = next(iter(acq.regions))
                self.region_selector.select_region(first_region)

        except Exception as e:
            self.log_panel.log(f"Error loading: {e}")

    def _on_region_selected(self, region_id: str) -> None:
        """Handle region selection: load frames into current view."""
        if self.controller.acquisition is None:
            return

        try:
            mode = self.controls_panel.current_view_mode()
            if mode == "mosaic":
                self._show_mosaic(region_id)
            else:
                self._show_fov(region_id, fov_index=0)
        except Exception as e:
            self.log_panel.log(f"Error: {e}")

    def _show_fov(self, region_id: str, fov_index: int = 0) -> None:
        """Display a single FOV in the viewer."""
        frame = self.controller.get_frame(region=region_id, fov=fov_index)

        if self._fov_viewer is None:
            from squid_tools.gui.viewer import SingleFOVViewer
            self._fov_viewer = SingleFOVViewer()
            self._viewer_stack.addWidget(self._fov_viewer)

        self._fov_viewer.display_frame(frame, name=f"FOV {fov_index}")
        self._viewer_stack.setCurrentWidget(self._fov_viewer)

    def _show_mosaic(self, region_id: str) -> None:
        """Display all FOVs for a region as a mosaic."""
        if self.controller.acquisition is None:
            return

        region = self.controller.acquisition.regions[region_id]
        frames = self.controller.get_region_frames(region=region_id)
        pixel_size = self.controller.acquisition.objective.pixel_size_um

        if self._mosaic_viewer is None:
            from squid_tools.gui.mosaic import MosaicView
            self._mosaic_viewer = MosaicView()
            self._viewer_stack.addWidget(self._mosaic_viewer)

        self._mosaic_viewer.load_region(region, frames, pixel_size)
        self._viewer_stack.setCurrentWidget(self._mosaic_viewer)

    def _on_view_mode_changed(self, mode: str) -> None:
        """Switch between FOV and mosaic view."""
        region_id = self.region_selector.selected_region_id()
        if region_id and self.controller.acquisition:
            if mode == "mosaic":
                self._show_mosaic(region_id)
            else:
                self._show_fov(region_id)

    def _on_borders_toggled(self, visible: bool) -> None:
        """Toggle FOV borders on mosaic."""
        if self._mosaic_viewer:
            self._mosaic_viewer.set_borders_visible(visible)

    def _on_run_requested(self, plugin_name: str, params: object) -> None:
        """Handle processing tab Run button."""
        if self.controller.acquisition is None:
            self.log_panel.log("No acquisition loaded")
            return

        region_id = self.region_selector.selected_region_id()
        if not region_id:
            self.log_panel.log("No region selected")
            return

        try:
            self.log_panel.set_status(f"Running {plugin_name}...")

            mode = self.controls_panel.current_view_mode()
            params_dict = params if isinstance(params, dict) else {}

            if mode == "mosaic":
                # Process all FOVs in region
                frames = self.controller.get_region_frames(region=region_id)
                for fov_idx, frame in frames.items():
                    result = self.controller.run_plugin(plugin_name, frame, params_dict)
                    fov = next(
                        f for f in self.controller.acquisition.regions[region_id].fovs
                        if f.fov_index == fov_idx
                    )
                    pixel_size = self.controller.acquisition.objective.pixel_size_um
                    y_px = fov.y_mm * 1000 / pixel_size
                    x_px = fov.x_mm * 1000 / pixel_size
                    if self._mosaic_viewer:
                        self._mosaic_viewer.display_processed(fov_idx, result, y_px, x_px)
            else:
                # Process single FOV
                frame = self.controller.get_frame(region=region_id, fov=0)
                result = self.controller.run_plugin(plugin_name, frame, params_dict)
                if self._fov_viewer:
                    self._fov_viewer.display_processed(result, name=f"{plugin_name} result")

            self.log_panel.set_status("Ready")
            self.log_panel.log(f"{plugin_name} complete")

        except Exception as e:
            self.log_panel.set_status("Error")
            self.log_panel.log(f"{plugin_name} failed: {e}")


def main() -> None:
    """Launch Squid-Tools standalone."""
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()
    sys.exit(app.exec_())


if __name__ == "__main__":
    main()
```

```python
"""Background subtraction plugin using sep (Source Extractor Python).

Estimates and subtracts spatially varying background from microscopy images.
"""

from __future__ import annotations

from typing import Any

import numpy as np
from pydantic import BaseModel

from squid_tools.core.data_model import Acquisition, OpticalMetadata
from squid_tools.plugins.base import ProcessingPlugin


class BackgroundParams(BaseModel):
    """Parameters for sep background estimation."""

    box_size: int = 64
    filter_size: int = 3


class BackgroundPlugin(ProcessingPlugin):
    """Background subtraction using sep.Background."""

    name = "Background Subtraction"
    category = "correction"
    requires_gpu = False

    def parameters(self) -> type[BaseModel]:
        return BackgroundParams

    def validate(self, acq: Acquisition) -> list[str]:
        return []

    def process(self, frames: np.ndarray, params: BaseModel) -> np.ndarray:
        import sep

        assert isinstance(params, BackgroundParams)
        data = frames.astype(np.float64, copy=True)
        # sep requires C-contiguous float64 array
        if not data.flags["C_CONTIGUOUS"]:
            data = np.ascontiguousarray(data)
        bkg = sep.Background(data, bw=params.box_size, bh=params.box_size,
                             fw=params.filter_size, fh=params.filter_size)
        return data - bkg.back()

    def default_params(self, optical: OpticalMetadata | None = None) -> BaseModel:
        return BackgroundParams()

    def test_cases(self) -> list[dict[str, Any]]:
        frame = np.full((128, 128), 100, dtype=np.float64)
        frame[50:70, 50:70] = 500
```

```python
"""Flatfield correction plugin.

Corrects illumination non-uniformity by dividing by a flatfield profile.
If no flatfield is provided, uses the image's own smoothed version as estimate.
"""

from __future__ import annotations

from typing import Any

import numpy as np
from pydantic import BaseModel, ConfigDict

from squid_tools.core.data_model import Acquisition, OpticalMetadata
from squid_tools.plugins.base import ProcessingPlugin


class FlatfieldParams(BaseModel):
    """Parameters for flatfield correction."""

    model_config = ConfigDict(arbitrary_types_allowed=True)

    flatfield: np.ndarray | None = None
    smoothing_sigma: float = 50.0


class FlatfieldPlugin(ProcessingPlugin):
    """Flatfield illumination correction."""

    name = "Flatfield Correction"
    category = "correction"
    requires_gpu = False

    def parameters(self) -> type[BaseModel]:
        return FlatfieldParams

    def validate(self, acq: Acquisition) -> list[str]:
        return []

    def process(self, frames: np.ndarray, params: BaseModel) -> np.ndarray:
        assert isinstance(params, FlatfieldParams)
        data = frames.astype(np.float64, copy=True)

        if params.flatfield is not None:
            flat = params.flatfield.astype(np.float64)
        else:
            # Estimate flatfield from the image itself using gaussian smoothing
            from scipy.ndimage import gaussian_filter
            flat = gaussian_filter(data, sigma=params.smoothing_sigma)

        # Normalize flatfield to mean 1
        flat_mean = np.mean(flat)
        if flat_mean > 0:
            flat = flat / flat_mean

        # Avoid division by zero
        flat = np.where(flat > 0.01, flat, 1.0)
        return data / flat

    def default_params(self, optical: OpticalMetadata | None = None) -> BaseModel:
        return FlatfieldParams()

    def test_cases(self) -> list[dict[str, Any]]:
        y, x = np.mgrid[0:128, 0:128]
        shading = 1.0 - 0.3 * ((x - 64) ** 2 + (y - 64) ** 2) / (64**2)
        frame = (1000 * shading).astype(np.float64)
        return [{"input": frame, "flatfield": shading, "description": "vignetting correction"}]
```
